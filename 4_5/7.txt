/*Продолжите программу. Объявите указатель на переменную a с именем ptr_a.
Через указатель ptr_a присвойте переменной a значение переменной b. Выведите в консоль значение переменной a в виде одного целого числа.*/

#include <stdio.h>

int main(void)
{
    short a, b;
    short *ptr_a; // Объявляем указатель на переменную типа short

    scanf("%hd", &b);

    ptr_a = &a; // Присваиваем указателю адрес переменной a
    *ptr_a = b; // Присваиваем переменной a значение переменной b через указатель

    printf("%hd\n", a); // Выводим значение переменной a в консоль

    __ASSERT_TESTS__ // макроопределение для тестирования (не убирать и должно идти непосредственно перед return 0)
    return 0;
}
