/* Продолжите программу. Объявите указатель ptr_a на переменную a и указатель ptr_b на переменную b.
Выполните обмен значениями переменных a и b через указатели ptr_a и ptr_b.
P. S. В консоль ничего выводить не нужно.*/

#include <stdio.h>

int main(void)
{
    short a, b, init_a, init_b;
    short *ptr_a, *ptr_b; // Объявляем указатели на переменные типа short

    scanf("%hd, %hd", &a, &b);
    init_a = a;
    init_b = b;

    ptr_a = &a; // Присваиваем указателю адрес переменной a
    ptr_b = &b; // Присваиваем указателю адрес переменной b

    short temp = *ptr_a; // Сохраняем значение переменной a во временной переменной
    *ptr_a = *ptr_b;     // Присваиваем переменной a значение переменной b через указатель ptr_a
    *ptr_b = temp;       // Присваиваем переменной b сохраненное ранее значение переменной a через указатель ptr_b

    __ASSERT_TESTS__ // макроопределение для тестирования (не убирать и должно идти непосредственно перед return 0)
    return 0;
}
