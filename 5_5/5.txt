/*Продолжите программу. Из массива pows необходимо удалить элемент с первым найденным четным значением.
Если такой элемент не был найден, то, соответственно, ничего удалять не нужно.
Выведите в консоль по порядку в одну строчку через пробел полученные значения оставшихся элементов массива pows.*/

#include <stdio.h>

#define TOTAL     20

int main(void)
{
    short pows[TOTAL] = {0};
    size_t count = 0;
    size_t sz_ar = sizeof(pows) / sizeof(*pows);

    while(count < sz_ar && scanf("%hd", &pows[count]) == 1)
        count++;

    // Поиск первого четного элемента и его удаление из массива
    for (size_t i = 0; i < count; ++i) {
        if (pows[i] % 2 == 0) {
            // Сдвиг элементов массива влево, начиная со следующего за найденным элементом
            for (size_t j = i; j < count - 1; ++j) {
                pows[j] = pows[j + 1];
            }
            // Уменьшение количества элементов в массиве
            count--;
            break;
        }
    }

    // Вывод оставшихся элементов массива
    for (size_t i = 0; i < count; ++i) {
        printf("%hd ", pows[i]);
    }
    printf("\n");

    return 0;
}
