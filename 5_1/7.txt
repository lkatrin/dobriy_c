/*В программе ниже объявите массив с именем func длиной 28 элементов типа double.
Прочитайте из входного потока вещественное число в переменную x и присвойте его седьмому элементу (седьмому по счету) массива func.
P. S. В консоль ничего выводить не нужно.*/

#include <stdio.h>

int main(void)
{
    // Объявление массива func длиной 28 элементов типа double
    double func[28];
    
    double x;
    
    // Чтение вещественного числа из входного потока в переменную x
    scanf("%lf", &x);
    
    // Присвоение введенного числа седьмому элементу (седьмому по счету) массива func
    func[6] = x;

    __ASSERT_TESTS__ // макроопределение для тестирования (не убирать и должно идти непосредственно перед return 0)
    return 0;
}
