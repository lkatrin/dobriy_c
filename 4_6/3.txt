/*  Продолжите программу. Объявите указатель ptr_a на переменную a и указатель ptr_b на переменную b.
Поменяйте адреса указателей ptr_a и ptr_b между собой, используя третий вспомогательный указатель с обобщенным типом void*.
Выведите в консоль в одну строчку через пробел целочисленные значения, на которые ссылаются указатели ptr_a и ptr_b именно в таком порядке:
<значение для ptr_a> <значение для ptr_b>*/

#include <stdio.h>

int main(void)
{
    short a, b;
    short *ptr_a, *ptr_b; // Объявляем указатели на переменные типа short
    void *temp_ptr; // Вспомогательный указатель с обобщенным типом

    scanf("%hd, %hd", &a, &b);

    ptr_a = &a; // Присваиваем указателю адрес переменной a
    ptr_b = &b; // Присваиваем указателю адрес переменной b

    temp_ptr = ptr_a; // Сохраняем адрес ptr_a во временном указателе
    ptr_a = ptr_b;    // Присваиваем ptr_a адрес ptr_b
    ptr_b = temp_ptr; // Присваиваем ptr_b сохраненный ранее адрес ptr_a

    printf("%d %d\n", *ptr_a, *ptr_b); // Выводим значения, на которые указывают ptr_a и ptr_b, через пробел

    __ASSERT_TESTS__ // макроопределение для тестирования (не убирать и должно идти непосредственно перед return 0)
    return 0;
}
