/*Продолжите программу. Необходимо из входного потока читать целые числа (по одному), пока не встретится число 78.
Все прочитанные числа по порядку заносить в массив ar, используя указатель ptr_ar (кроме последнего числа 78).
Вывести в консоль прочитанные значения из массива ar в одну строчку через пробел в порядке их чтения.*/

#include <stdio.h>

int main(void)
{
    int ar[100];
    const int * const ptr_ar = ar;
    
    int num;
    int count = 0;
    int *write_ptr = ar; // Указатель для записи в массив ar

    // Чтение целых чисел из входного потока до встречи числа 78
    while (scanf("%d", &num) == 1 && num != 78) {
        *write_ptr++ = num; // Запись числа в массив и сдвиг указателя записи
        count++; // Увеличение счетчика
    }

    // Вывод прочитанных значений из массива ar в одну строку через пробел
    for (int i = 0; i < count; ++i) {
        printf("%d ", ar[i]);
    }
    printf("\n");

    return 0;
}
