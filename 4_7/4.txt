/*Допишите следующую программу. Необходимо все 8 байт (тип double) переменной value вывести в консоль в виде десятичных чисел,
записанных в одну строчку через пробел, начиная с первого байта.*/

#include <stdio.h>

int main(void)
{
    double value;

    scanf("%lf", &value);

    char *ptr = (char *)&value; // Приводим указатель на double к указателю на char для доступа к каждому байту

    for (int i = 0; i < sizeof(double); i++) {
        printf("%d ", (int)ptr[i]); // Выводим каждый байт в виде десятичного числа, разделенные пробелами
    }
    printf("\n"); // Переходим на новую строку

    return 0;
}
