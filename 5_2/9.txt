/*В программе до функции main() объявлен массив с именем marks и типом элементов short.
Этот массив скрыт в коде, но он существует.
В функции main() вычислите среднее арифметическое значений элементов этого массива
и выведите в консоль полученное значение с точностью до сотых.*/

#include <stdio.h>

int main(void)
{
    int num_elements = sizeof(marks) / sizeof(marks[0]); // Вычисление количества элементов в массиве

    int sum = 0; // Переменная для хранения суммы значений элементов массива

    // Вычисление суммы значений элементов массива
    for (int i = 0; i < num_elements; i++) {
        sum += marks[i];
    }

    // Вычисление среднего арифметического
    float average = (float)sum / num_elements;

    // Вывод с точностью до сотых
    printf("%.2f\n", average);

    return 0;
}
