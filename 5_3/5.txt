/*Продолжите программу. Прочитайте из входного потока целое число в переменную value типа int.
Объявите массив с именем ar_d типа char и длиной для побайтного хранения числа типа int.
Скопируйте значение каждого байта переменной value в массив ar_d.
Выведите значения элементов массива ar_d в консоль в виде десятичных чисел в одну строчку через пробел.*/

#include <stdio.h>

int main(void) {

    int value;

    // Чтение целого числа из входного потока
    scanf("%d", &value);

    // Объявление массива для побайтного хранения числа типа int
    char ar_d[sizeof(int)];

    // Копирование значений каждого байта переменной value в массив ar_d
    for (size_t i = 0; i < sizeof(int); ++i) {		// Цикл будет выполняться, пока значение i меньше размера типа int в байтах
        ar_d[i] = (value >> (i * 8)) & 0xFF;
    }

    // Вывод значений элементов массива ar_d в консоль
    for (size_t i = 0; i < sizeof(int); ++i) {
        printf("%d ", ar_d[i]);
    }
    printf("\n");

    __ASSERT_TESTS__ // макроопределение для тестирования (не убирать и должно идти непосредственно перед return 0)
    return 0;
}
